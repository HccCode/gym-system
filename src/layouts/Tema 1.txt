import { Outlet, Link, useLocation } from 'react-router-dom';
import { useGymStore } from '../store/useGymStore';

export default function Layout({ usuarioActual, onLogout }) {
  const location = useLocation();
  const { configuracion } = useGymStore();

  // Nuevo estilo de botón activo: Gradiente Índigo a Púrpura con sombra neón
  const isActive = (path) => location.pathname === path 
    ? "bg-gradient-to-r from-indigo-600 to-violet-600 text-white shadow-lg shadow-indigo-500/25" 
    : "text-zinc-400 hover:bg-zinc-800/50 hover:text-white";

  const puedeVer = (permiso) => usuarioActual.permisos?.includes(permiso);

  return (
    // 1. Fondo general negro profundo con un poco de padding (p-4) y espacio (gap-4)
    <div className="flex h-screen bg-[#09090b] p-4 gap-4 font-sans text-zinc-100 print:bg-white print:p-0">
      
      {/* 2. SIDEBAR FLOTANTE: Bordes muy redondeados y efecto cristal */}
      <aside className="w-64 bg-zinc-900/40 backdrop-blur-2xl border border-zinc-800/60 rounded-[2rem] flex flex-col relative z-20 print:hidden overflow-hidden shadow-2xl">
        
        <div className="p-6 border-b border-zinc-800/60 flex flex-col items-center justify-center gap-4 min-h-[6rem]">
          {configuracion?.logo && (
            <img 
              src={configuracion.logo} 
              alt="Logo Gym" 
              className="max-h-14 max-w-[12rem] object-contain drop-shadow-lg" 
            />
          )}
          <h1 className="text-xl font-black text-white tracking-widest text-center uppercase">
            {configuracion?.nombreGym || 'GYM'}
          </h1>
        </div>
        
        <nav className="flex-1 px-4 py-6 space-y-2 overflow-y-auto custom-scrollbar">
          {usuarioActual.rol === 'recepcionista' && puedeVer('pos') && (
            <Link to="/pos" className={`block px-4 py-3 rounded-xl transition-all font-medium ${isActive('/pos')}`}>🛒 Punto de Venta</Link>
          )}

          {puedeVer('dashboard') && <Link to="/" className={`block px-4 py-3 rounded-xl transition-all font-medium ${isActive('/')}`}>📊 Dashboard</Link>}
          {puedeVer('kiosco') && <Link to="/checkin" className={`hidden md:block xl:hidden px-4 py-3 rounded-xl transition-all font-medium ${isActive('/checkin')}`}>📷 Kiosco Acceso</Link>}
          {puedeVer('miembros') && <Link to="/miembros" className={`block px-4 py-3 rounded-xl transition-all font-medium ${isActive('/miembros')}`}>👥 Miembros</Link>}
          
          {usuarioActual.rol !== 'recepcionista' && puedeVer('pos') && (
            <Link to="/pos" className={`block px-4 py-3 rounded-xl transition-all font-medium ${isActive('/pos')}`}>🛒 Punto de Venta</Link>
          )}

          {puedeVer('suscripciones') && <Link to="/suscripciones" className={`block px-4 py-3 rounded-xl transition-all font-medium ${isActive('/suscripciones')}`}>💳 Suscripciones</Link>}
          {puedeVer('inventario') && <Link to="/inventario" className={`block px-4 py-3 rounded-xl transition-all font-medium ${isActive('/inventario')}`}>📦 Inventario</Link>}
          {puedeVer('caja') && <Link to="/caja" className={`block px-4 py-3 rounded-xl transition-all font-medium ${isActive('/caja')}`}>💵 Corte de Caja</Link>}
          
          {puedeVer('ajustes') && (
            <div className="pt-4 mt-4 border-t border-zinc-800/60">
              <Link to="/ajustes" className={`block px-4 py-3 rounded-xl transition-all font-medium ${isActive('/ajustes')}`}>⚙️ Ajustes</Link>
            </div>
          )}
        </nav>
        
        <div className="p-4 bg-zinc-950/30 text-xs font-black tracking-widest text-center text-zinc-500 uppercase">
          GYM<span className="text-indigo-500">SYSTEM</span>
        </div>
      </aside>

      <main className="flex-1 flex flex-col gap-4 overflow-hidden relative z-10 print:overflow-visible">
        
        {/* 3. HEADER FLOTANTE: En forma de píldora (rounded-full) */}
        <header className="bg-zinc-900/40 backdrop-blur-2xl border border-zinc-800/60 h-16 rounded-[1.5rem] flex items-center justify-between px-6 shadow-lg print:hidden">
          <h2 className="text-lg font-bold text-zinc-100 capitalize">
            Panel de {usuarioActual.rol}
          </h2>
          
          <div className="flex items-center gap-4 bg-zinc-950/50 pl-2 pr-4 py-1.5 rounded-full border border-zinc-800/60">
            <div className="w-8 h-8 rounded-full flex items-center justify-center text-white font-bold shadow-lg bg-gradient-to-tr from-indigo-500 to-violet-500 shadow-indigo-500/20">
              {usuarioActual.nombre.charAt(0).toUpperCase()}
            </div>
            <span className="text-sm font-semibold text-zinc-300">
              {usuarioActual.nombre}
            </span>
            <div className="w-px h-4 bg-zinc-700"></div>
            <button onClick={onLogout} className="text-sm font-bold text-rose-500 hover:text-rose-400 transition-colors flex items-center gap-1.5">
              <span>🔒</span> Bloquear
            </button>
          </div>
        </header>

        {/* 4. CONTENEDOR PRINCIPAL FLOTANTE */}
        <div className="flex-1 overflow-y-auto bg-zinc-900/20 backdrop-blur-sm border border-zinc-800/40 rounded-[2rem] p-8 shadow-inner print:p-0 print:border-none print:bg-white">
          <Outlet context={{ rol: usuarioActual.rol }} /> 
        </div>
      </main>
    </div>
  );
}
